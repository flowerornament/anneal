    const TERMINAL_STATUS_HEURISTICS: &[&str] = &[
        "superseded", "archived", "retired", "deprecated",
        "obsolete", "withdrawn", "cancelled", "canceled",

    2. When `previous_snapshot` is `Some`, use temporal signal instead of static edge analysis. Replace the stall detection logic:

    For each ordering level with >= 3 handles:
    - If `previous_snapshot` is Some: look up the status name for this level from `lattice.ordering[level_idx]`, get its count from `previous_snapshot.states`. If current count >= previous count, it's a stall.
    - If `previous_snapshot` is None: fall back to existing static edge-based detection (check for DependsOn outflow to next level).

    The implementation should be:
    ```rust
    if handles_at_level.len() < 3 {
        continue;
    }

    let status_name = &lattice.ordering[level_idx];
    let is_stall = if let Some(prev) = previous_snapshot {
        let prev_count = prev.states.get(status_name).copied().unwrap_or(0);
        let current_count = handles_at_level.len();
        current_count >= prev_count && prev_count >= 3
    } else {
        // Fallback: static analysis -- no DependsOn edges to next level
        let next_level = level_idx + 1;
        !handles_at_level.iter().any(|&node_id| {
            graph.edges_by_kind(node_id, EdgeKind::DependsOn).any(|edge| {
                let target = graph.node(edge.target);
                if let Some(ref target_status) = target.status {
                    lattice::state_level(target_status, lattice) == Some(next_level)
                } else {
                    false
                }
            })
        })
    };
    ```

    Update the S003 message to indicate the signal source:
    - Temporal: "pipeline stall at '{status_name}': {count} handles (unchanged from previous snapshot)"
    - Static: keep existing message format
